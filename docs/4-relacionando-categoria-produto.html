🎯 Módulo 1: O que muda com o relacionamento

Regra de negócio: uma Categoria pode ter muitos Produtos, e cada produto pertence a uma categoria.

Tipo da relação: 1:N (um para muitos).

1️⃣ Categoria -> N Produtos 🔑 Chave estrangeira em Produto 🧭 API retorna categoria junto do produto

🗂️ Módulo 2: Atualizar schema completo

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
}

model Categoria {
  id           Int       @id @default(autoincrement())
  nome         String    @unique
  descricao    String
  criadoEm     DateTime  @default(now())
  atualizadoEm DateTime  @updatedAt
  produtos     Produto[]
}

model Produto {
  id           Int        @id @default(autoincrement())
  nome         String     @unique
  quantidade   Int
  preco        Float
  criadoEm     DateTime   @default(now())
  atualizadoEm DateTime   @updatedAt
  categoriaId  Int?
  categoria    Categoria? @relation(fields: [categoriaId], references: [id])
}

🖥️ Módulo 4: Ajustes no frontend

// Tipos com categoria
interface Categoria {
  id: number;
  nome: string;
  descricao: string;
}

interface Produto {
  id: number;
  nome: string;
  quantidade: number;
  preco: number;
  categoriaId: number | null;
  categoria?: { id: number; nome: string } | null;
}

// Estados novos
const [categorias, setCategorias] = useState<Categoria[]>([]);
const [mostrarFormCategoria, setMostrarFormCategoria] = useState(false);
const [novaCategoria, setNovaCategoria] = useState({ nome: "", descricao: "" });
const [erroCategoria, setErroCategoria] = useState<string | null>(null);

const [formData, setFormData] = useState({
  nome: "",
  quantidade: 0,
  preco: 0,
  categoriaId: "",
});

// Carregar categorias
const carregarCategorias = async () => {
  const res = await fetch("/api/categoria");
  const data = await res.json();
  setCategorias(data);
};

// Criar categoria sem sair do modal
const handleCriarCategoriaRapida = async () => {
  const res = await fetch("/api/categoria", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(novaCategoria),
  });

  const criada = await res.json();
  setCategorias((prev) => [criada, ...prev]);
  setFormData((prev) => ({ ...prev, categoriaId: String(criada.id) }));
  setNovaCategoria({ nome: "", descricao: "" });
  setMostrarFormCategoria(false);
};

// Select de categoria no modal de produto
<div>
  <label className="mb-1 block text-lg font-medium text-gray-700">Categoria</label>
  <div className="flex gap-2">
    <select
      className="w-full rounded-lg border border-blue-300 p-2"
      value={formData.categoriaId}
      onChange={(e) => setFormData({ ...formData, categoriaId: e.target.value })}
    >
      <option value="">Sem categoria</option>
      {categorias.map((categoria) => (
        <option key={categoria.id} value={String(categoria.id)}>
          {categoria.nome}
        </option>
      ))}
    </select>
    <button type="button" onClick={() => setMostrarFormCategoria((prev) => !prev)}>
      Nova
    </button>
  </div>
</div>

// Formulário rapido de nova categoria
{mostrarFormCategoria && (
  <div className="rounded-lg border border-blue-200 bg-blue-50/60 p-3">
    <input placeholder="Nome da categoria" value={novaCategoria.nome} onChange={(e) => setNovaCategoria({ ...novaCategoria, nome: e.target.value })} />
    <input placeholder="Descrição" value={novaCategoria.descricao} onChange={(e) => setNovaCategoria({ ...novaCategoria, descricao: e.target.value })} />
    {erroCategoria && <p>{erroCategoria}</p>}
    <button type="button" onClick={handleCriarCategoriaRapida}>Salvar categoria</button>
  </div>
)}

// Envio do produto com categoriaId
const payload = {
  ...formData,
  categoriaId: formData.categoriaId ? Number(formData.categoriaId) : null,
};

// Exibir categoria no card
<p className="text-sm text-gray-600">
  Categoria: <span className="font-semibold text-gray-900">{produto.categoria?.nome ?? "Sem categoria"}</span>
</p>
